pipeline {
    agent any
    tools {
        maven 'M1'
    }
    stages {
        stage('Fetching') {
            steps {
                git branch: 'main', url: "https://github.com/SarishtiG/java.git"
            }
        }
        stage('Compile') {
            steps {
                bat 'mvn compile'
            }
        }
        stage('Test') {
            steps {
                bat 'mvn test'
            }
        }
    }
}
